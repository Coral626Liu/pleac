Reformatting Paragraphs
s.wrap(width) =
    ls, s' = s.words.foldl(([], ""), ((ls, s), word ->
        if "{s} {word}" > width then
            ls + [s], word
        else
            s' = (s &&& "{s} ") + word
            ls, s'
    ))
    join(ls + [s'], "\n")
