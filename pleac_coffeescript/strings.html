cut2fmt = (positions) ->
    template   = ''
    lastpos    = 1
    for place in positions
        template += "A" + (place - lastpos) + " "
        lastpos   = place
    template += "A*"

fmt = cut2fmt([8, 14, 20, 26, 30])
console.log fmt
# A7 A6 A6 A6 A4 A*
